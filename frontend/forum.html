io.on('connection', (socket) => { console.log('User connected'); socket.on('join-chat', (username) => { socket.join('general-chat'); socket.username = username; // Update online users const onlineUsers = []; for (let [id, socket] of io.sockets.sockets) { if (socket.username) onlineUsers.push(socket.username); } io.emit('online-users', onlineUsers); }); socket.on('chat-message', (data) => { io.to('general-chat').emit('chat-message', data); }); socket.on('disconnect', () => { const onlineUsers = []; for (let [id, socket] of io.sockets.sockets) { if (socket.username) onlineUsers.push(socket.username); } io.emit('online-users', onlineUsers); }); });
